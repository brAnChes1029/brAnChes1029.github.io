#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n;
int a[200005];
int b[200005];
unordered_map<int,int> p;

int main(){

    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;
    for( int i=1 ; i<=n ; i++ ){
        cin >> a[i];
        p[a[i]]=i;
    }

    int cnt=0;
    cin >> b[1];
    for( int i=2 ; i<=n ; i++ ){
        cin >> b[i];
        if( p[b[i-1]]!=p[b[i]]-1 ) cnt++;
        //cout << p[b[i-1]] << ' ' << p[b[i]] << '\n';
    }

    cout << cnt;

    return 0;
}

那就按照上面的策略寫吧 ~ 

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll const mod=1e9+7;
ll ans[16]={1, 1, 2, 3, 4, 6, 9, 12, 18, 27, 36, 54, 81, 108, 162, 243};

ll pw( ll x, ll y ){
    if( y==0 ) return 1;
    else if( y==1 ) return x;
    else{
        ll z=pw( x, y/2 );
        if( y&1 ) return z*z%mod*x%mod;
        else return z*z%mod;
    }
}

int main(){

    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int t;
    cin >> t;
    while( t-- ){
        ll x;
        cin >> x;
        if( x<=15 ) cout << ans[x] << '\n';
        else{
            ll y=x/6;
            ll z=x%6;
            if( z!=0 ){
                y--;
                z+=6;
            }
            cout << pw( ans[6], y )*ans[z]%mod << '\n';
        }
    }

    return 0;
}

暴雷

一開始就先字串處理嘛，會不會他的考點其實是這個，然後從有賦值的儲存格開始拓樸，就做完了。
那個 ana 是 analysis 的意思喔 （￣ ︶ ￣）

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, m;
bool vis[200][30];
int ind[200005];
int v[200][30];
vector<int> a[200005];

void ana( string s, int i, int j ){
    int k=0;
    while( s[k]!='+' ) k++;
    if( s[1]>='A' && s[1]<='Z' ){
        int f=stoi(s.substr(2, k-2))*100+s[1]-'A'+1;
        a[f].push_back(i*100+j);
        ind[i*100+j]++;
    }
    else{
        v[i][j]+=(s[1]-'0');
    }
    k++;
    if( s[k]>='A' && s[k]<='Z' ){
        int f=stoi(s.substr(k+1))*100+s[k]-'A'+1;
        a[f].push_back(i*100+j);
        ind[i*100+j]++;
    }
    else{
        v[i][j]+=(s[k]-'0');
    }
}

int main(){

    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n >> m;
    queue<int> q;
    for( int i=1 ; i<=n ; i++ ){
        for( int j=1 ; j<=m ; j++ ){
            string s;
            cin >> s;
            if( s.length()==1 ){
                v[i][j]=(s[0]-'0');
                q.push( i*100+j );
                vis[i][j]=true;
            }
            else if( s.length()==4 ){
                v[i][j]=(s[1]-'0'+s[3]-'0');
                q.push( i*100+j );
                vis[i][j]=true;
            }
            else{
                ana( s, i, j );
            }
        }
    }

    while( !q.empty() ){
        int x=q.front();
        q.pop();
        int i=x/100;
        int j=x%100;
        for( int y : a[x] ){
            ind[y]--;
            v[y/100][y%100]+=v[i][j];
            if( ind[y]==0 ){
                vis[y/100][y%100]=true;
                q.push(y);
            }
        }
    }

    for( int i=1 ; i<=n ; i++ ){
        for( int j=1 ; j<=m ; j++ ){
            if( !vis[i][j] ) cout << "ERROR ";
            else cout << v[i][j] << ' ';
        }
        cout << '\n';
    }

    return 0;
}
