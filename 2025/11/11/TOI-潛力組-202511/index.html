#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, x;
int dp[10005];

int main(){

    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n >> x;
    for( int i=1 ; i<10005 ; i++ ) dp[i]=1e9;

    for( int i=0 ; i<n ; i++ ){
        int a, b;
        cin >> a >> b;
        b+=a;
        for( int j=b ; j<10005 ; j++ ){
            dp[j]=min( dp[j], dp[j-b]+a );
        }
    }

    int ans=1e9;
    for( int i=x ; i<10005 ; i++ ) ans=min( ans, dp[i] );
    cout << ans;

    return 0;
}

P1

暴雷

如果有 $n$ 個人帽子同一個顏色，那這 $n$ 個人都會看到 $n-1$。
如果有超過 $n$ 個人看到 $n-1$，那就代表要分多組。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n;
int c[1005];

int main(){

    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;
    for( int i=0 ; i<n ; i++ ){
        int x;
        cin >> x;
        c[x+1]++;
    }
    int all=0;
    for( int i=1 ; i<1005 ; i++ ){
        int y=(c[i]/i);
        if( c[i]%i ) y++;
        all+=i*y;
    }
    cout << all;

    return 0;
}

P3

暴雷

先把圖分成很多個邊雙連通分量，如果剪斷的邊不是橋那就沒有影響。
把雙連通分量們和橋們組成樹，然後就可以在上面 DP，就能知道剪斷某個橋會影響到下面多少個子樹。
實作一樣先找出我的模板先找出橋 (dfs) ，再把雙連通分量編號 (dfs2) ，最後再 DP (dfs3)。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, m, q, c;
vector<int> a[50005];
vector<int> b[50005]; // BCC
set< pair<int,int> > bridge;
int color[50005];
int dep[50005];
int low[50005];
int sz[50005];
int dp[50005];

void dfs( int i, int p ){
    for( int u : a[i] ){
        if( u==p ) continue;
        if( !dep[u] ){
            dep[u]=dep[i]+1;
            low[u]=dep[u];
            dfs( u, i );
            low[i]=min( low[i], low[u] );
            if( low[u]>dep[i] ) bridge.insert({min(i,u), max(i,u)});
        }
        else{
            low[i]=min( low[i], dep[u] );
        }
    }
}

void dfs2( int i, int p ){
    sz[c]++;
    color[i]=c;
    for( int u : a[i] ){
        if( u==p || color[u] ) continue;
        if( bridge.count({min(i,u), max(i,u)}) ) continue;
        dfs2( u, i );
    }
}

void dfs3( int i, int p ){
    dp[i]=sz[i];
    for( int u : b[i] ){
        if( u==p ) continue;
        dfs3( u, i );
        dp[i]+=dp[u];
    }
}

int main(){

    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n >> m >> q;
    for( int i=0 ; i<m ; i++ ){
        int x, y;
        cin >> x >> y;
        a[x].push_back(y);
        a[y].push_back(x);
    }

    dep[0]=low[0]=1;
    dfs( 0, -1 );

    /*
    cout << bridge.size() << '\n';
    for( auto u : bridge ){
        cout << u.first << ' ' << u.second << '\n';
    }
    */

    for( int i=0 ; i<n ; i++ ){
        if( color[i] ) continue;
        c++;
        dfs2( i, -1 );
    }
    for( auto [x, y] : bridge ){
        b[color[x]].push_back(color[y]);
        b[color[y]].push_back(color[x]);
    }
    dfs3( 1, -1 );

   while( q-- ){
        int x, y;
        cin >> x >> y;
        if( x>y ) swap(x, y);
        if( bridge.count({x, y}) ){
            cout << n-min( dp[color[x]], dp[color[y]] ) << '\n';
        }
        else{
            cout << n << '\n';
        }
   }

    return 0;
}
